 324     /**
 325      * Mostra o Valor no float Formatado
 326      *
 327      * @param  string  $number
 328      * @param  integer $decimals
 329      * @param  boolean $showThousands
 330      * @return string
 331      */
 332     public static function nFloat($number, $decimals = 2, $showThousands = false)
 333     {
 334         if (is_null($number) || empty(self::onlyNumbers($number)) || floatval($number) == 0) {
 335             return 0;
 336         }
 337         $pontuacao = preg_replace('/[0-9]/', '', $number);
 338         $locale = (mb_substr($pontuacao, -1, 1) == ',') ? "pt-BR" : "en-US";
 339         $formater = new \NumberFormatter($locale, \NumberFormatter::DECIMAL);
 340 
 341         if ($decimals === false) {
 342             $decimals = 2;
 343             preg_match_all('/[0-9][^0-9]([0-9]+)/', $number, $matches);
 344             if (!empty($matches[1])) {
 345                 $decimals = mb_strlen(rtrim($matches[1][0], 0));
 346             }
 347         }
 348 
 349         return number_format($formater->parse($number, \NumberFormatter::TYPE_DOUBLE), $decimals, '.', ($showThousands ? ',' : ''));
 350     }

 574             return $digito;
 575         }
 576         return $sum%11;

 581      *
 582      * @return int
 583      */
 584     public static function modulo10($n)
 585     {
 586         $chars = array_reverse(str_split($n, 1));
 587         $odd = array_intersect_key($chars, array_fill_keys(range(1, count($chars), 2), null));
 588         $even = array_intersect_key($chars, array_fill_keys(range(0, count($chars), 2), null));
 589         $even = array_map(
 590             function ($n) {
 591                 return ($n >= 5) ? 2*$n - 9 : 2*$n;
 592             }, $even
 593         );
 594         $total = array_sum($odd) + array_sum($even);
 595         return ((floor($total/10) + 1)*10 - $total)%10;
 596     }
