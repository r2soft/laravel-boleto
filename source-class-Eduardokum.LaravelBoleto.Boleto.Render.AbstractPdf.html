139      * BarCode
140      *
141      * @param     $xpos

145      * @param int $height

149     public function i25($xpos, $ypos, $code, $basewidth = 1, $height = 10)
150     {
151         $code = (strlen($code)%2 != 0 ? '0' : '') . $code;
152         $wide = $basewidth;
153         $narrow = $basewidth/3;
154 
155         $barChar = array();
156         // wide/narrow codes for the digits
157         $barChar['0'] = 'nnwwn';
158         $barChar['1'] = 'wnnnw';
159         $barChar['2'] = 'nwnnw';
160         $barChar['3'] = 'wwnnn';
161         $barChar['4'] = 'nnwnw';
162         $barChar['5'] = 'wnwnn';
163         $barChar['6'] = 'nwwnn';
164         $barChar['7'] = 'nnnww';
165         $barChar['8'] = 'wnnwn';
166         $barChar['9'] = 'nwnwn';
167         $barChar['A'] = 'nn';
168         $barChar['Z'] = 'wn';
169 
170         $this->SetFont('Arial', '', 10);
171         $this->SetFillColor(0);
172 
173         // add start and stop codes
174         $code = 'AA' . strtolower($code) . 'ZA';
175 
176         for ($i = 0; $i < strlen($code); $i = $i + 2) {
177             // choose next pair of digits
178             $charBar = $code[$i];
179             $charSpace = $code[$i + 1];
180             // check whether it is a valid digit
181             if (!isset($barChar[$charBar])) {
182                 $this->Error('Invalid character in barcode: ' . $charBar);
183             }
184             if (!isset($barChar[$charSpace])) {
185                 $this->Error('Invalid character in barcode: ' . $charSpace);
186             }
187             // create a wide/narrow-sequence (first digit=bars, second digit=spaces)
188             $seq = '';
189             for ($s = 0; $s < strlen($barChar[$charBar]); $s++) {
190                 $seq .= $barChar[$charBar][$s] . $barChar[$charSpace][$s];
191             }
192             for ($bar = 0; $bar < strlen($seq); $bar++) {
193                 // set lineWidth depending on value
194                 if ($seq[$bar] == 'n') {
195                     $lineWidth = $narrow;
196                 } else {
197                     $lineWidth = $wide;
198                 }
199                 // draw every second value, because the second digit of the pair is represented by the spaces
200                 if ($bar%2 == 0) {
201                     $this->Rect($xpos, $ypos, $lineWidth, $height, 'F');
202                 }
203                 $xpos += $lineWidth;
204             }
205         }
206     }
